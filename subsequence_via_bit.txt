#include<bits/stdc++.h>
using namespace std;
void filter(int n,char arr[]){
int j=0;
while(n>0){
    int last_bit=(n&1);
    if(last_bit){
        cout<<arr[j];
    }
    n=n>>1;
    j++;
}
cout<<endl;

}
void print(char arr[]){
    int x=strlen(arr);
    for(int i=0;i<(1<<x);i++){
    filter(i,arr);
    }
}
int main(){
    char arr[200]="abc";
    print(arr);
}
