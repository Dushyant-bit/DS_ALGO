//Program to Show BFS Traversal
#include<bits/stdc++.h>
using namespace std;
class Graph{
map<int,list<int>> MAP;
public:
void addEdge(int x,int y){
    MAP[x].push_back(y);
    MAP[y].push_back(x);
}
void traverse(int src){
    map<int ,int>visited;
    queue<int> traverser;
    traverser.push(src);
    visited[src]=true;
    while(!traverser.empty()){
        int node=traverser.front();
        cout<<node<<" ";
         traverser.pop();
        for(auto x : MAP[node]){
            if(!visited[x]){
                 traverser.push(x);
                  visited[node]=true;
             }
        }
    }
}
};
int main(){
Graph g;
g.addEdge(0,1);
g.addEdge(1,2);
g.addEdge(2,3);
g.addEdge(3,4);
g.addEdge(4,5);
g.traverse(0);
return 0;
}
