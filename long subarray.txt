#include<bits/stdc++.h>
using namespace std;
int divisible(int const *arr,int k){
    int q=sizeof(arr);
    map<int ,int>MAP;
MAP[0]=1;
int sum=0,ans=0;
for(int i=0;i<q;i++){
    sum+=arr[i];
    int rem=sum%k;
    if(rem<0){
        rem=rem+k;
    }
    if(MAP.find(rem)!=MAP.end()){
        ans+=MAP[rem];
        MAP[rem]++;
    } else{
        MAP[rem]=1;
    }
}
return ans;
}
int main(){

    int arr[]={2,-6,3,1,2,8,2,1};
    int k=7;
    cout<<divisible(arr,k);
}
