#include<bits/stdc++.h>
using namespace std;
int dp[6][6]={0};
int cost[6][6]={{1,5,2},{7,1,1},{8,1,3}};
int mincostovergrid(int R,int C){
    for(int i=0;i<R;i++){
        for(int j=0;j<C;j++){
        if(i==0 and j==0){
            dp[i][j]=cost[i][j];
        }
        else if(i==0){
        dp[i][j]=dp[i][j-1]+cost[i][j];
            }
        else if(j==0){
            dp[i][j]=dp[i-1][j]+cost[i][j];
        }
        else{
            dp[i][j]=min(dp[i-1][j],dp[i][j-1])+cost[i][j];
            }
        }

    }
    return dp[R-1][C-1];
}
int main(){
cout<<mincostovergrid(3,3);
}
