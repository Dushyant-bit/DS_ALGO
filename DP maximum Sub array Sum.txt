//DP maximum Sub array Sum
// In linear Time O(n)
#include<bits/stdc++.h>
using namespace std;
int compute(int const *arr ,int n){
int dp[100]={0};
dp[0]=arr[0]<0?0:arr[0];
int maxsum=0;
for(int i=1;i<n;i++){
    dp[i]=arr[i]+dp[i-1];
    if(dp[i]<0){
        dp[i]=0;
    }
    maxsum=max(dp[i],maxsum);

}
return maxsum;
}
int main(){
    int n[]={-3,2,5,-1,6,3,-2,7,-5,2};
    int si=sizeof(n)/sizeof(int);
    cout<<compute(n,si);
}
