#include<bits/stdc++.h>
using namespace std;
//Points To Revise
//Ladders Bottom -Up Approach Just Counting All Possible Path
//Time complexity O(NK  )
int Ladder_Bottom_Up(int *dp,int k,int n) {
    int ways=0;
for(int j=3;j<=n;j++) {
    for (int i = 1; i <= k; i++) {
        if (j - i >= 0) {
            dp[j]+=dp[j-i];
        }
    }
}
    return dp[n];
}
int main(){
    int dp[100]={0};
    dp[0]=1;
    dp[1]=1;
    dp[2]=2;
    int n=4,k=3;
    cout<< Ladder_Bottom_Up(dp,k,n);

}
